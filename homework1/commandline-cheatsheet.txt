# 1. create directory temp/
mkdir temp/


# 2. change working directory to temp/ (assuming temp/ exists)
cd temp/ && pwd
# check current working directory 
pwd 


# 3. write 'hello world' to file called hello.txt
echo 'hello world' > hello.txt


# 4. list all files and folders in the parent folder of temp/
ls ../


# 5. list all currently processes running
ps -e 


# 6. kill process with ID 666
kill 666


# 7. write the contents of environment variable $PATH to file path.txt 
echo $PATH > path.txt


# 8. find file 'hello.txt' in the parent of the current directory 
find ../ -name 'hello.txt' 


# 9. find all .png files in your home directory and subdirectories
find ./ -name '*.png' 


# 10. find where the executable of command echo is located 
whereis echo 


# 11. find text 'Select all processes' in the manual of command ps (that is, in the output of command man ps) 
man ps | grep 'Select all processes' 


# 12. write the last 20 commands you've executed to file commands.txt 
history 20 > commands.txt
